    function updateDisplay() {
        const now = new Date();
        
        // Update Jam (Format 00:00:00)
        const hours = String(now.getHours()).padStart(2, &#39;0&#39;);
        const minutes = String(now.getMinutes()).padStart(2, &#39;0&#39;);
        const seconds = String(now.getSeconds()).padStart(2, &#39;0&#39;);
        document.getElementById(&#39;clock&#39;).innerText = `${hours}:${minutes}:${seconds}`;

        // Update Tanggal Indonesia
        const options = { weekday: &#39;long&#39;, year: &#39;numeric&#39;, month: &#39;long&#39;, day: &#39;numeric&#39; };
        document.getElementById(&#39;current-date&#39;).innerText = now.toLocaleDateString(&#39;id-ID&#39;, options).toUpperCase();
    }

    setInterval(updateDisplay, 1000);
